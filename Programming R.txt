# Генерація випадкових даних - Поліщук, Десик, Бойчук УМ-11
set.seed(42)  # Фіксуємо генератор випадкових чисел для відтворюваності
x <- rnorm(100, mean = 5, sd = 2)  # 100 значень з нормального розподілу, середнє 5, стандартне відхилення 2
y <- 2 * x + rnorm(100, mean = 0, sd = 3)  # Залежність y = 2x + випадкова похибка

# Побудова графіку
plot(x, y, 
     main = "Графік залежності між x та y",
     xlab = "x - незалежна змінна",
     ylab = "y - залежна змінна",
     col = "blue", 
     pch = 16)  # Сині точки

# Додавання трендової лінії
model <- lm(y ~ x)  # Лінійна регресія y залежить від x
abline(model, col = "red", lwd = 2)  # Лінія регресії на графіку

# Виведення параметрів моделі
summary(model)
